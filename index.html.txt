    <script>
        const horizonElement = document.getElementById('horizon-element');
        const pitchScale = document.getElementById('pitch-scale');
        const startBtn = document.getElementById('start-btn');
        const container = document.getElementById('ai-container');
        const info = document.getElementById('info');

        // Pitch Ölçeğini Oluştur
        function createPitchScale() {
            for (let deg = -40; deg <= 40; deg += 5) {
                if (deg === 0) continue;
                const isMajor = deg % 10 === 0;
                const line = document.createElement('div');
                line.className = `pitch-line ${isMajor ? 'major' : 'minor'}`;
                const yPos = 400 - (deg * 5); 
                line.style.top = `${yPos}px`;

                if (isMajor) {
                    const labelL = document.createElement('div');
                    labelL.className = 'pitch-label';
                    labelL.innerText = Math.abs(deg);
                    labelL.style.top = `${yPos - 7}px`;
                    labelL.style.left = `calc(50% - 60px)`;
                    pitchScale.appendChild(labelL);

                    const labelR = document.createElement('div');
                    labelR.className = 'pitch-label';
                    labelR.innerText = Math.abs(deg);
                    labelR.style.top = `${yPos - 7}px`;
                    labelR.style.left = `calc(50% + 45px)`;
                    pitchScale.appendChild(labelR);
                }
                pitchScale.appendChild(line);
            }
        }
        createPitchScale();

        // Sensör İşleme Fonksiyonu
        function handleMotion(event) {
            let roll = event.gamma;  // Sağ-sol yatış
            let pitch = event.beta;  // Ön-arka eğim

            if (roll === null || pitch === null) return;

            // Ufuk göstergesi mantığı: Cihaz öne eğilirse (pitch artar), mavi gökyüzü aşağı iner.
            // Bu yüzden pitch değerini dikey kaydırma (translateY) olarak kullanıyoruz.
            let pitchOffset = pitch * 5; 
            
            horizonElement.style.transform = `rotate(${-roll}deg) translateY(${pitchOffset}px)`;
            
            info.innerText = `Pitch: ${Math.round(pitch)}° | Roll: ${Math.round(roll)}°`;
        }

        // Başlatma ve İzin Alma Butonu
        startBtn.addEventListener('click', () => {
            // iOS 13+ için izin isteme
            if (typeof DeviceOrientationEvent.requestPermission === 'function') {
                DeviceOrientationEvent.requestPermission()
                    .then(permissionState => {
                        if (permissionState === 'granted') {
                            window.addEventListener('deviceorientation', handleMotion);
                            startBtn.style.display = 'none';
                            container.style.display = 'block';
                        } else {
                            alert("Sensör izni reddedildi.");
                        }
                    })
                    .catch(console.error);
            } else {
                // Android veya eski iOS
                window.addEventListener('deviceorientation', handleMotion);
                startBtn.style.display = 'none';
                container.style.display = 'block';
            }
        });
    </script>
